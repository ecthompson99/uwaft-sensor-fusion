140  template <typename BusType, typename BusInfoType, typename MsgType>
141  inline int setReceivedAndCurrentLengths(const BusType &busProp, BusInfoType &busInfoProp,
142  const MsgType &msgProp, const slros::ROSPropertyWarnStatus &warnStatus)
143  {
144  const int numItemsReceived = msgProp.size();
145  const int maxBusArraySize = slros::getNumItemsInFixedSimpleArray(busProp);
146  const int numItemsToCopy = std::min(numItemsReceived, maxBusArraySize);
147 
148  busInfoProp.ReceivedLength = numItemsReceived;
149  busInfoProp.CurrentLength = numItemsToCopy;
150 
151  if (numItemsReceived > maxBusArraySize)
152  {
153  warnStatus.emitTruncationWarning(numItemsReceived, maxBusArraySize);
154  }
155  return numItemsToCopy;
156  }
