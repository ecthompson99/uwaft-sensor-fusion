172 template <class CppParamType, class ROSCppParamType>
173 uint8_t SimulinkParameterGetter<CppParamType,ROSCppParamType>::get_parameter(CppParamType* dataPtr)
174 {
175  XmlRpc::XmlRpcValue xmlValue;
176  ROSCppParamType paramValue;
177  bool paramRetrieved = false;
178 
179  // Get parameter as XmlRpcValue and then parse it through our own function
180  if (nodePtr->getParam(paramName, xmlValue))
181  {
182  paramRetrieved = param_parser::getScalar(xmlValue, paramValue);
183  }
184 
185  // Cast the returned value into the data type that Simulink is expecting
186  *dataPtr = static_cast<CppParamType>(paramValue);
187 
188  const uint8_t errorCode = process_received_data(dataPtr, paramRetrieved);
189  return errorCode;
190 }

199 template <class CppParamType, class ROSCppParamType>
200 uint8_t SimulinkParameterGetter<CppParamType,ROSCppParamType>::process_received_data(CppParamType* dataPtr, bool paramRetrieved)
201 {
202  // By default, assume that parameter can be retrieved successfully
203  uint8_t errorCode = errorCodeSuccess;
204 
205  if (!paramRetrieved)
206  {
207  // An error code of "errorCodeNoParam" means that the parameter does not exist and
208  // "errorCodeTypeMismatch" means that it exists, but the data types don't match
209  errorCode = nodePtr->hasParam(paramName) ? errorCodeTypeMismatch : errorCodeNoParam;
210  }
211 
212  if (errorCode == errorCodeSuccess)
213  {
214  // Remember last valid value
215  lastValidValue = *dataPtr;
216  hasValidValue = true;
217  }
218  else
219  {
220  // An error occurred. If a valid value was previously
221  // received, return it. Otherwise, return the
222  // initial value.
223  if (hasValidValue)
224  *dataPtr = lastValidValue;
225  else
226  *dataPtr = initialValue;
227  }
228 
229  return errorCode;
230 }

314 template <class CppParamType, class ROSCppParamType>
315 uint8_t SimulinkParameterArrayGetter<CppParamType,ROSCppParamType>::process_received_data(
316  const ROSCppParamType& retrievedValue, const uint32_t maxLength, bool paramRetrieved,
317  CppParamType* dataPtr, uint32_t* receivedLength)
318 {
319 
320  // By default, assume that parameter can be retrieved successfully
321  uint8_t errorCode = errorCodeSuccess;
322 
323  if (!paramRetrieved)
324  {
325  // An error code of "errorCodeNoParam" means that the parameter does not exist and
326  // "errorCodeTypeMismatch" means that it exists, but the data types don't match
327  errorCode = nodePtr->hasParam(paramName) ? errorCodeTypeMismatch : errorCodeNoParam;
328  }
329 
330  if (errorCode == errorCodeSuccess)
331  {
332  // Indicate truncation if received array has more elements than maxLength
333  if (retrievedValue.size() > maxLength)
334  errorCode = errorCodeArrayTruncate;
335 
336  // Copy the received data into the Simulink variable location
337  // We don't need to do zero-padding here, because GetParameterArrayState::codegenStepImpl
338  // initializes dataPtr to all zeros.
339  uint32_t copyLength = std::min(maxLength, static_cast<uint32_t>(retrievedValue.size()));
340  ROS_ASSERT(copyLength <= maxLength);
341  std::copy(retrievedValue.begin(), retrievedValue.begin() + copyLength, dataPtr);
342  *receivedLength = copyLength;
343 
344  // Remember last valid value
345  lastValidValue.resize(copyLength);
346  std::copy(retrievedValue.begin(), retrievedValue.begin() + copyLength, lastValidValue.begin());
347  hasValidValue = true;
348  }
349  else
350  {
351  // An error occurred. If a valid value was previously
352  // received, return it. Otherwise, return the
353  // initial value.
354  if (hasValidValue)
355  {
356  // lastValidValue is truncated when it is saved, so there is no possibility
357  // of a buffer overrun in dataPtr here
358  ROS_ASSERT(lastValidValue.size() <= maxLength);
359  std::copy(lastValidValue.begin(), lastValidValue.begin() + lastValidValue.size(), dataPtr);
360  *receivedLength = uint32_t(lastValidValue.size());
361  }
362  else
363  {
364  // initialValue is truncated in GetParameterArrayState::setupSetInitialValue,
365  // so there is no possibility of a buffer overrun in dataPtr here
366  ROS_ASSERT(initialValue.size() <= maxLength);
367  std::copy(initialValue.begin(), initialValue.begin() + initialValue.size(), dataPtr);
368  *receivedLength = uint32_t(initialValue.size());
369  }
370  }
371 
372  return errorCode;
373 }
